path属性 #

大写表示绝对单位, 小写表示相对单位 d属性

M 起始点的坐标, 添加多个M的时候表示重新绘制新的路径
L 结束点的坐标
基本绘制
Z 表示自动闭合, 在最后出现, 前面的闭合
H 水平绘制
Z 垂直绘制

<path d="M500 1200H200V200" stroke="black" stroke-width="5"></path>

动态创建svg标签 #

var svgNs = 'http://www.w3.org/2000/svg';
var oParent = document.querySelector('.svg-con');
// 创建svg标签
var oSvg = document.createElementNS(svgNs, 'svg');
oSvg.setAttribute('xml', svgNs);
oSvg.setAttribute('width', '100%');
oSvg.setAttribute('height', '100%');
oParent.appendChild(oSvg);

// 封装创建标签方法
function createTag(tag, objAttr) {
    var oTag = document.createElementNS(svgNs, tag);
    for (var attr in objAttr) {
        oTag.setAttribute(attr, objAttr[attr]);
    }
    return oTag
}
let tag = createTag('line', {
    'x1': '10',
    'y1': '10',
    'x2': 200,
    'y2': 200,
    'stroke': '#ccc',
    'stroke-width': '3'
})
oSvg.appendChild(tag);

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <title>Document</title>

    <style>

        * {

            padding: 0;
            margin: 0;
            box-sizing: border-box;
        }

        .father {
            width: 300px;
            height: 800px;
            background-color: pink;
            border: 1px solid #ccc;
        }
    </style>
</head>

<body>
    <div class="svg-con">
        <div class="tit"></div>
        <div class="con"></div>
    </div>
    <script>
        var svgNs = 'http://www.w3.org/2000/svg';
        var oParent = document.querySelector('.svg-con');
        // 创建svg标签
        var oSvg = document.createElementNS(svgNs, 'svg');
        oSvg.setAttribute('xml', svgNs);
        oSvg.setAttribute('width', '100%');
        oSvg.setAttribute('height', '100%');
        oParent.appendChild(oSvg);

        // 封装创建标签方法
        function createTag(tag, objAttr) {
            var oTag = document.createElementNS(svgNs, tag);
            for (var attr in objAttr) {
                oTag.setAttribute(attr, objAttr[attr]);
            }
            return oTag
        }
        let tag = createTag('line', {
            'class': 'svgline',
            'x1': '10',
            'y1': '10',
            'x2': 200,
            'y2': 200,
            'stroke': '#ccc',
            'stroke-width': '3'
        })
        oSvg.appendChild(tag);

        let oPolyline, pointsNum, oLine;
        let preDot = {
            x: 0,
            y: 0
        }; // 上一个点
        oSvg.onmousedown = function(ev) {
            var ev = ev || window.event;
            if (!oPolyline) {
                // 第一次创建标签
                oPolyline = createTag('polyline', {
                    fill: 'none',
                    'stroke': 'red',
                    'stroke-width': '2'
                })
                oSvg.appendChild(oPolyline)
            }
            // 算出当前鼠标点击的点在osvg中的那个位置
            var x = ev.clientX - oParent.offsetLeft;
            var y = ev.clientY - oParent.offsetTop;
            preDot = {
                x,
                y
            };
            // 判断是否是开始点
            if (!pointsNum) {
                pointsNum = x + ',' + y
            } else {
                pointsNum = pointsNum + ',' + x + ',' + y
            }
            console.log("pointsNum", pointsNum)

            oPolyline.setAttribute('points', pointsNum)
        }

        oSvg.onmousemove = function(ev) {
            console.log("=======>", '移动');
            var ev = ev || window.event;
            // 算出当前鼠标点击的点在osvg中的那个位置
            var x = ev.clientX - oParent.offsetLeft;
            var y = ev.clientY - oParent.offsetTop;
            // 移动的时候, 绘制线条
            if (!oLine) {
                oLine = createTag('line', {
                    fill: 'none',
                    'stroke': 'red',
                    'stroke-width': '2'
                })
                oSvg.appendChild(oLine)
            }
            if (preDot.x === 0 && preDot.y === 0) {
                return
            }
            // , y1: preDot.y, x2: x, y2: y }
            oLine.setAttribute('x1', preDot.x)
            oLine.setAttribute('y1', preDot.y)
            oLine.setAttribute('x2', x)
            oLine.setAttribute('y2', y)

        }
    </script>
</body>

</html>
