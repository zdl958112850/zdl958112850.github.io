  5. 常用 api

  • gulp.task() 语法: gulp.task(任务名称, 任务处理函数) 例子:
gulp.task("htmlHandler", function () {});
  • gulp.src() 语法: gulp.src(路径信息) =>作用：找到源文件 =>书写方式 2-1.gulp.src（./a/b.htm1） ->找到指定一个文件 2-2.gulp.src（'./a/*.html'）

    找到指定目录下，指定后缀的文件 2-3.gulp.src（'./a/’） 找到指令目录下的所有文件 2-4.gulp.src（'./a//1） ->找到 a 目录下所有子目录里面的所有文件 2-5.gulp.src（'./a/**/.html’） ->找到 a 目录下所有子目录里面的所有.html 文件

  • gulp.dest（） =>语法：gulp.dest（路径信息） =>作用：把一个内容放入指定目录内 =>例子：gulp.dest（'./abc'） ->把他接收到的内容放到 abc 目录下
  • gulp.watch（） =>语法：gulp.watch（路径信息，任务名称） =>作用：监控指定目录下的文件，一旦发生变化，从新执行后面的任务 =>例子:gulp.watch('./src/pages/*.html',htmlHandler) ->当指定目录下的 html 文件发生变化,就会执行 htmlHandler 这个任务
  • gulp.series（） =>语法：gulp.series（任务 1，任务 2，任务 3，.…） =>作用：逐个执行多个任务，前一个任务结束，第二个任务开始
  • gulp.parallel（） =>语法：gulp.parallel（任务 1，任务 2，任务 3，.） =>作用：并行开始多个任务
  • pipe（） =>管道函数 =>所有的 gulpAPI 都是基于流 =>接收当前流，进入下一个流过程的管道函数 gulp.src().pipe(压缩任务).pipe(转码)

// gulp@4
const autoPrefixer = require("gulp-autoprefixer");
const sass = require("gulp-sass");
const uglify=require("gulp-uglify")
const babel = require('gulp-babel')
const htmlmin = require('gulp-htmlmin')

const cssHandler = function () {
  return gulp
    .src("./src/css/*.css")
    .pepe(
      autoPrefixer({
        browers: ["last 2 versions"], // 兼容最近两个版本的浏览器
      })
    )
    .pipe(cssmin()) //1-2.压缩文件
    .pipe(gulp.dest("./dist/css/"));
};

//1-2.创建一个打包sass 文件的在务
const sassHandler=function(){
    return gulp
    .src('./src/sass/*.scss')
    .pipe(sass())
    ·pipe(autoprefixer())
    .pipe(cssmin())
    .pipe(gulp.dest("./dist/sass/"))
}

//1-3.创建一个打包js文件的在务, 不能压缩es6语法, 需要先转为es5
const jsHandler=function(){
    return gulp
        .src("./src/js/*.js")//1.龙到js文件
        .pipe(babel({
            presets: ['@babel/env']
        }))
        .pipe(uglify())
        ·pipe(gulp.dest('./dist/js/'))
}

//1-4.创建一个打包html文件的在务
const htmlHandler=function(){
    return gulp
    .src('./src/pages/*.htm1')
    .pipe(htmlmin({
        collapseWhitespace: true, // 移除空格
    }))
    .pipe(gulp.dest("./dist/pages/"))
}

gulp配置代理

var gulp = require('gulp');
var webserver = require('gulp-webserver');

gulp.task('webserver', function () {
  gulp.src('./')
    .pipe(webserver({
      host: 'localhost',
      port: 8888,
      livereload: true, // 实时重新加载
      open: './index.html', // 启动时默认浏览器打开的文件
      directoryListing: {
        enable: true,
        path: './'
      },
      proxies: [{
        source: '/api',
        target: 'http://www.your.com' // 代理的域名
      }]
    }))
});
// gulp4.0
gulp.task('default', gulp.series(['webserver'], function () {
  console.log('gulp webserver start successfully at localhost:8888');
}));
// 因为gulp升级到了4.0，所以使用gulp.series来启动，如果使用的是gulp3.*的话请将上面的代码替换为下面的
// gulp 3.*
gulp.task('default', ['webserver'], function () {
  console.log('gulp webserver start successfully at localhost:8888');
});
