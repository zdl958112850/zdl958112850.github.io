写这篇博客目的 #

昨天配置了一个雷达图, 各种配置都需要慢慢去找效果, 再去找改的那个属性, 实属麻烦 解决了你想改这块, 但不知道这块叫啥名字的尴尬

通用配置 #

雷达图 #

配置都写在这里面

radar: {

},

center 居中位置 [百分比或数字] #

radius 大小 百分比或数字 #

startAngle 旋转 数字 #

axisName 每一条轴的名称 {} #

axisName: {
  show: true,
  formatter: function (value, indicator) {
    // value 当前indicator的name, indicator当前哪行数据, 可以加data
    return '【' + value + '】' + ': ' + indicator.data;
  },
  color: '#333',
  lineHeight: '2'
},

nameGap 轴名称距离轴的距离 number #

nameGap: 100

splitNumber 分割数量 number #

splitNumber: 10

shape 形状 string #

shape: 'circle' , // polygon

axisLine 轴线 object #

axisLine: {
  show: true,
  symbol: 'arrow', // none, arrow, ['arrow', 'none'] 一边有箭头
  symbolSize: [10, 15], // 箭头的宽高
  symbolOffset: [0, 0], // 箭头的偏移
}

lineStyle 控制线的样式 object

axisLine: {
  show: true,
  lineStyle: {
      color: '#333',
      width: 1,
      type: 'solid',
    }
}

axisTick 轴上刻度 object #

axisTick: {
  show: true,
  length: 15,
  // lineStyle: {} 同上
}

axisLabel 刻度上的数字 object #

axisLabel: {
  show: true,
  rotate: 0, // 刻度的文字旋转角度
  margin: 8,
  // formatter ... ,
  // showMinLabel ... ,
  // showMaxLabel ... ,
  // hideOverlap ... ,
  // color ... ,
  fontStyle: 'normal',
  fontWeight: 'normal',
  fontFamily: 'sans-serif',
  fontSize: 12,
  // align ... ,
  // verticalAlign ... ,
  // lineHeight ... ,
  backgroundColor: 'transparent',
  // borderColor ... ,
  borderWidth: 0,
  borderType: 'solid',
  borderDashOffset: 0,
  borderRadius: 0,
  padding: 0,
  shadowColor: 'transparent',
  shadowBlur: 0,
  shadowOffsetX: 0,
  shadowOffsetY: 0,
  width: 20 ,
  // height ... ,
  // textBorderColor ... ,
  // textBorderWidth ... ,
  textBorderType: 'solid',
  textBorderDashOffset: 0,
  textShadowColor: 'transparent',
  textShadowBlur: 0,
  textShadowOffsetX: 0,
  textShadowOffsetY: 0,
  overflow: 'truncate', // truncate 截断 需要配置 width并且配置截断后展示ellipsis, break, breakAll
  ellipsis: '...' ,
}

splitLine 分割线 object #

 splitLine: {
  show: false,
  lineStyle: {
    color: ['#ccc'],
    width: 1,
    type: 'solid',
    dashOffset: 0,
    cap: 'butt',
    join: 'bevel',
    miterLimit: 10,
    shadowOffsetX: 0,
    shadowOffsetY: 0,
    opacity: 1
  }
}

splitArea 分割区域的样式 #

splitArea: {
  show: true,
  areaStyle: {
    color: ['rgba(250,250,250,0.3)', 'rgba(255,200,200,0.3)'],
    shadowOffsetX: 0,
    shadowOffsetY: 0,
    opacity: 1
  }
}

indicator 指示器 array #

indicator: [
  { name: '销售（sales）', max: 6500, min: 4000 , data: 333333 }, 最低4000, 最大6500, 中间进行分割
]

饼图 #

配置都在 option 的 series 里面

option = {
  series: [
    {
      type: "pie",
      radius: "50%",
      data: [],
  ],
};

radius 控制饼图大小 %,number,string,array #

  • 百分比（数字，字符串同）

  • 数组 数组里面的具体值跟单个配置一样

minAngle 最小散区角度 number [0-360] #

avoidLabelOverlap 标签重叠策略 boolean #

label 文字指示配置 {} #

配置指示文字样式及文字过长自动换行

label: {
  formatter(v) {
    let text = v.name;
    if (text.length <= 6) {
      return `{b|${text}}\n\n{c|${v.value}}\n`;
    } else if (text.length > 6 && text.length <= 10) {
      return (text = `{b|${text.slice(0, 6)}}\n\n{b|${text.slice(
        6
      )}}\n\n{c|${v.value}}\n`);
    }
  },
  rich: {
    b: {
      lineHeight: 10,
      align: "left",
      fontSize: 16,
      fontFamily: "SimHei",
    },
    c: {
      lineHeight: 10,
      align: "center",
      fontSize: 22,
      color:"#ccc",
      fontFamily: "SimHei",
    },
  },
},

roseType 南丁格尔配置 string #

可选值：'radius' | 'area'

  • radius:扇区圆心角按数据百分比展示，半径展示数据大小
  • area:扇区圆心角都一致

配置技巧 #

  1. 去它官网找一个 demo, 改它配置, 而且是自动保存更新的, 改好配置后再复制到项目中去
  2. 直接复制它配置项的默认值, 然后再去改
