        count.value += 1;
        console.log(count);

        // 在页面挂在的时候
    });
    return {
        count,
        handleCountAdd,
    };
},

  • watch
watchEffect(() => {
    console.log(user.type); // 当user.type改变的时候会触发该函数
})

// 单独监听
watch(num, (newNum, prevNum) => {
    console.log(newNum, prevNum)
    console.log('当num改变时候，会触发执行此函数')
})

//多个监听
watch([num, user], (newNum, prevNum) => {
    console.log(newNum, prevNum)
    console.log(当num改变时候， 会触发执行此函数)
})
